# use _update to add/remove or change a document field using doc or script commands

POST books/_update/1
{

}

# add new field
POST books/_update/1
{
  "script" : "ctx._source.name = 'ameera'"
}

# remove a field
POST books/_update/1
{
  "script" : "ctx._source.remove('name')"
}

# update a field
POST books/_update/1
{
  "script" : "ctx._source.isbn = 'HIJ'"
}
